-- We could either have explicit matching

defn lastInArray :: [a] -> a {
    match [a] {
        [] = error "No end for empty array"
        [x] = x
        [_:tail] =  lastInArray tail
    }
}


defn lastInList :: (a) -> a {
    () = error "No end for empty list"
    (x) = x
    (_:tail) =  lastInList tail
}
